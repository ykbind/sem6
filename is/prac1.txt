1.	Write a C++ program that contains a string (char pointer) with a value \Hello World’. The program should AND or and XOR each character in this string with 127 and display the result.

#include <iostream>
using namespace std;
int main()
 {
    // Define string using char pointer

    char *str = (char*)"Hello World";
    cout << "Original String: " << str << endl;
    cout << "\nAfter AND with 127:\n";
    for (int i = 0; str[i] != '\0'; i++) {
        char result = str[i] & 127;
        cout << result;
    }
    cout << "\n\nAfter XOR with 127:\n";
    for (int i = 0; str[i] != '\0'; i++) {
        char result = str[i] ^ 127;
        cout << result;
    }

   cout << endl;
    return 0;
}
